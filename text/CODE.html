// KODELIFE #version 150 uniform float time; uniform vec2 resolution; uniform vec2 mouse; in VertexData{ vec4 v_position; vec3 v_normal; vec2 v_texcoord; } inData; out vec4 fragColor; vec4 Shader(vec2 p, vec2 mouse){ p -= vec2(.1,0.); p *= mat2(1.05,0.,0.,1.05); vec4 col = vec4(1.); float r = (mouse.x*2.+.1)*sin(p.x*15.)*cos(15.*p.y); r = sin(7.*r+time*2.); float d = length(p*.5); col *= smoothstep(r,r+.1,d); return col;} void main(void){ vec2 uv = (gl_FragCoord.xy-.5*resolution.xy)/resolution.y; vec4 col = Shader(uv, mouse); fragColor = col;}
